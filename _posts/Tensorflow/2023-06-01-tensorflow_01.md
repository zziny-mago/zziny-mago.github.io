~~~
    t1=tf.Variable([1,2,3])
    t2=tf.constant([1,2,3])

    print(type(t1))
    print(type(t2))
~~~

~~~
    t1=tf.constant([1,2,3])
    t2=tf.Variable([1,2,3])
    #t3=tf.constant(t2)
    t4=tf.Variable(t1)

    print(type(t4))
~~~

~~~
    t1=tf.convert_to_tensor([1,2,3])
    t2=tf.Variable([1,2,3])
    t3=tf.convert_to_tensor(t2)

    print(type(t1))
    print(type(t3))
~~~

~~~
    t1=tf.Variable([1,2,3])
    t2=tf.constant([4,5,6])
    t3=tf.Variable([1,2,3])
    t4=tf.constant([4,5,6])

    print(type(t2+t4))
    print(type(t1+t2))
    print(type(t1+t3))
~~~
